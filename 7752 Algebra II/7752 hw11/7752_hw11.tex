\begin{homeworkProblem}
  In this problem you will need the following two definitions. \\

  \textbf{Definition 1:} Let $L/F$ be a finite separable extension and let $\overline{F}$ be an algebraic closure of $F$ containing $L$. A subfield $L'$ of $\overline{F}$ is called \textbf{conjugate to $L$ over $F$} if $L'=\sigma(L)$ for some $F$-embedding $\sigma:L\to\overline{F}$. (Note: $L/F$ is Galois if and only if the only conjugate to $L$ over $F$ is itself.) \\

  \textbf{Definition 2:} A finite extension $K/F$ is called a \textbf{$p$-extension} if $K/F$ is \textbf{Galois} and $\Gal(K/F)$ is a $p$-group.\\

  \textbf{(a)}: Let $L/F$ be a separable extension of degree $n$ and let $K$ be the Galois closure of $L$ over $F$. Prove that $K$ can be written as a compositum $L_1 L_2\cdots L_n$, where $L_1,\ldots, L_n$ are (not necessarily distinct) conjugates of $L$ over $F$.

  \begin{proof}
    By the primitive element theorem, there exists some $\alpha\in L$ such that $L = F(\alpha)$. Let $\alpha_1,\ldots,\alpha_n$ be the roots of $\mu_{\alpha,F}$ (all inside $K$ as $K/F$ is Galois). By the simple extension lemma, for each $i\in\{1,\ldots, n\}$ the inclusion $F\sub \cls{F}$ extends to an $F$-embedding $\sigma_i:F(\alpha)\hookrightarrow \cls{F}$ such that $\sigma_i(\alpha) = \alpha_i$. Now set $L_i = \sigma_i (L) = \sigma_i(F(\alpha))$.\\

    Then $L_1\cdots L_n = F(\alpha_1,\ldots,\alpha_n)$ is a splitting field for the separable polynomial $\mu_{\alpha,F}$ over $F$, whence $L_1\cdots L_n/F$ is Galois. Thus by minimality of the Galois closure, $K\sub L_1\cdots L_n$. On the other hand, as $\alpha_1,\alpha_n\in K$, it follows that $L_1\cdots L_n = F(\alpha_1,\ldots,\alpha_n)\sub K$. Thus $K = L_1\cdots L_n$.
  \end{proof}

  \textbf{(b)}:Let $K/F$ and $L/F$ be finite $p$-extensions. Prove that $KL/F$ is also a $p$-extension.

  \begin{proof}
    By homework 10 problem 4 part (ii), $KL/F$ is finite Galois and we have an injective group homomorphism $\iota:\Gal(KL/F)\hookrightarrow\Gal(K/F)\times\Gal(L/F)$. As both $\Gal(K/F)$ and $\Gal(L/F)$ are finite $p$-groups, it follows then that $\Gal(KL/F)\cong \iota(\Gal(KL/F))\sub \Gal(K/F)\times\Gal(L/F)$ is a finite $p$-group.
  \end{proof}

  \textbf{(c)}: Suppose that $K/L$ and $L/F$ are both $p$-extensions, and let $M$ be the Galois closure of $K$ over $F$ (note: we do not know whether $K/F$ is Galois or not). Prove that $M/F$ is also a $p$-extension.

  \begin{proof}
    By part (a), $M = K_1\cdots K_n$ for some conjugates $K_i$ of $K$ over $F$. Then for $i\in \{ 1,\ldots,n\}$, there is some $F$-embedding $\sigma_i:K\hookrightarrow\cls{F}$ such that $K_i = \sigma_{i}(K)$. By normality of $L/F$, $\sigma_i(L)=L$. Thus each extension $K_i/L$ is $F$-isomorphic to $K/L$ via $\sigma_i$ and is thus a $p$-extension. Now by part (b), $M/L = K_1\cdots K_n / L$ is a $p$-extension, whence we observe that
    \[
      |\Gal(M/F)| = [M:F] = [M:L][L:F] = |\Gal(M/L)|\cdot|\Gal(L/F)|
    \]
    implies that $M/F$ is also a $p$-extension.
  \end{proof}

  \textbf{(d)}: Now assume only that $L/F$ is a separable extension with $[L:F]=p^r$, for some $r\geq 1$. Let $M$ be the Galois closure of $L$ over $F$. Prove that $[M:F]$ need not be a power of $p$.

  \begin{proof}
    Consider the example $L/F = \Q(\sqrt[p^r]{2})/\Q$. Then $M = \Q(\sqrt[p^r]{2}, \zeta_{p^r})$ where $\zeta_{p^r}$ is a primitive $p^r$th root of unity. On one hand $[\Q(\zeta_{p^r}):\Q] = \phi(p^r) = p^r - p^{r-1} = p^{r-1}(p-1)/2$ is not a power of $p$. On the other hand, $p^{r-1}(p-1)/2 = [\Q(\zeta_{p^r}):\Q] \divides [M:\Q]$, so it follows that $[M:\Q]$ is not a power of $p$.
  \end{proof}

\end{homeworkProblem}



\begin{homeworkProblem}
  Let $f(x)$ and $g(x)$ be irreducible polynomials in $\F_p[x]$ of the same degree. Let $F=\F_p[x]/(f(x))$. Prove that $g(x)$ splits completely over $F$.

  \begin{proof}
    By a vector space counting argument, $|F| = p^n$. By uniqueness of splitting fields, $F$ is $\F_p$-isomorphic to $\F_{p^n}$ which is $\F_p$-isomorphic to $\F_p[x]/(q(x))$ which contains a root of $q(x)$. Thus, $F$ contains a root of $q(x)$ whence by normality of the extensions $F/\F_p$, $q(x)$ splits over $F$.
  \end{proof}
\end{homeworkProblem}

  \textbf{(a)}: Determine the Galois group $G$ of the splitting field $K$ of $f(x)$ over $\Q$.

  \begin{proof}
    Let $\alpha = \sqrt{1+\sqrt{6}}$ and $\beta = \sqrt{1-\sqrt{6}}$. Then $f(x) = (x-\alpha)(x+\alpha)(x-\beta)(x+\beta)$ and $K = \Q(\alpha,\beta)$. Noting that $\alpha^2+\beta^2 = 2$, it follows that $\mu_{\beta,\Q(\alpha)} = x^2 +(\alpha^2-2)$ and thus $[K:\Q(\alpha)] = 2$. Note that $f(x)$ is irreducible as  none of the choices of pairs of linear factors provide a polynomial in $\Q[x]$ by appealing to Vieta's formulae and the fact that $\alpha^2,\beta^2,\alpha\pm\beta\not\in\Q$. Hence, $[\Q(\alpha): \Q] = 4$ and thus $|G| = [K:\Q] = 8$.\\

    Letting $\alpha_1 = \alpha$, $\alpha_2 = -\alpha$, $\alpha_3 = \beta$, $\alpha_4 = -\beta$, it follows that the action of $G$ on $\{\alpha_1,\cdots,\alpha_4\}$ induces an injective group homomorphism $\rho:G\hookrightarrow S_4$. Thus $G\cong\rho(G)\sub S_4$ is an order $8$ subgroup of $S_4$, all of which are isomorphic to $D_4$ so $G\cong D_4$.

\begin{homeworkProblem}
  Let $p$ and $q$ be distinct primes with $q>p$, and let $K/F$ be a Galois extension of degree $pq$. Prove the following:\\

  \textbf{(a)}: There exists a field $L$ with $F\subset L\subset K$ and $[L:F]=q$.

  \begin{proof}
    Let $G = \Gal(K/F)$. Then $|G| = pq$, whence by Sylow's existence theorem there is some subgroup $H\sub G$ such that $|H| = p$. Setting $L = K^{H}$, by the fundamental theorem of Galois theory, $p = |H| = [K:K^H]$ whence $[K^H: F] = q$ as desired.
  \end{proof}

  \textbf{(b)}: There exists a \textbf{unique} field $M$ with $F\subset M\subset K$ and $[M:F]=p$.

  \begin{proof}
    Let $G = \Gal(K/F)$. Let $n_q$ denote the number of Sylow $q$-subgroups of $G$. Then as $n_q \divides p$ and $n_q\equiv 1 mod\ q$, the restriction that $q>p$ forces  $n_q = 1$. Thus there is a unique subgroup of $Q$ of $G$ of order $q$, whence by the fundamental theorem of Galois theory there is a unique intermediate subfield $M = K^Q$ of $K/F$ with $[K:M] = q$ or equivalently $[M:F] = p$
  \end{proof}

\end{homeworkProblem}


\begin{homeworkProblem}
  Prove the following analogue of K\"{u}mmer's theorem for abelian extensions: Let $n\in\N$ and let $F$ be a field containing a primitive $n^{th}$ root of unity.\\

  \textbf{(a)}: Let $K/F$ be a finite Galois extension such that $G=\Gal(K/F)$ is abelian of exponent $n$. Then there exists $a_1,\ldots,a_t\in F$ such that $K=F(\sqrt[n]{a_1},\ldots,\sqrt[n]{a_t})$. More precisely, there exists $\alpha_1,\ldots,\alpha_t\in K$ such that $K=F(\alpha_1,\ldots,\alpha_t)$ and $\alpha_i^n\in F$ for all $i$.

  \begin{proof}
    Write $G \cong C_{n_1}\times\cdots C_{n_t}$ where $n_i\in\N$ and $C_{n_i} = \Z/n_{i}\Z$. For $j\in\{1,\ldots, t\}$, set $G_j = \prod_{i\neq j}C_{n_i}$ and $K_j = K^{G_j}$. As $G_j\triangleleft G$, it follows that $K_j/F$ is Galois and $\Gal(K_j/F)\cong \Gal(K/F)/\Gal(K/K_j) = G/G_j\cong C_{n_j}$. Hence, by K\"{u}mmer's theorem, there exists some $\alpha_j \in K_j$ such that $K_j = F(\alpha_j)$ and $\alpha_j^{n_j}\in F$, whence $\alpha_j^n = (\alpha_j^{n_j})^{n/n_j} \in F$.\\

    % TODO maybe justify fixed field of intersection of subgroups is compositum
    Observe that
    \[
      K = K^{\{e\}} = K^{\bigcap_{i=1}^{t} G_i} = K_1 K_2 \cdots K_t = F(\alpha_1, \alpha_2,\cdots, \alpha_t).
    \]
  \end{proof}

  \textbf{(b)}: Conversely, suppose that $K=F(\sqrt[n]{a_1},\ldots,\sqrt[n]{a_t})$ for some $a_1,\ldots, a_t\in F$. Prove that $K/F$ is Galois and $G=\Gal(K/F)$ is abelian of exponent $n$. \textbf{Hint:} For part (b) use one of the problems from the previous homework.

  \begin{proof}
    Write $\alpha_i = \sqrt[n]{a_i}$, so $\alpha_i^n = a_i$. As $F$ contains a primitive $n^{th}$ root of unity, K\"{u}mmer's theorem implies that each $F(\alpha_i)/F$ is Galois with $\Gal(F(\alpha_i)/F)\cong \Z/n_i\Z$ for some $n_i\divides n$. Applying homework 10 problem 4(b) part(ii) iteratively, we obtain an embedding
    \[
      \iota: \Gal(K/F)\hookrightarrow \prod_{i=1}^{t}\Gal(F(\alpha_i)/F) \cong \prod_{i=1}^{t}\Z/n_i\Z
    \]
    As the latter group has exponent dividing $n$, it follows that $\Gal(K/F)$ has exponent dividing $n$.
  \end{proof}
\end{homeworkProblem}


\begin{homeworkProblem}
  Let $F$ be a field containing a primitive $n^{th}$ root of unity. Let $a,b\in F$ be such that the polynomials $f(x)=x^n-a$, and $g(x)=x^n-b$ are both irreducible over $F$. Consider the K\"{u}mmer extensions $F(\alpha)$, $F(\beta)$, where $\alpha$ is a root of $f(x)$ and $\beta$ is a root of $g(x)$. Prove that
  $F(\alpha)=F(\beta)$ if and only if $\beta=c\alpha^r$, for some $c\in F$ and some integer $r$ which is coprime to $n$ (equivalently, if and only if $b=c^n a^r$, for some $c\in F$ and some $(r,n)=1)$.

  \begin{proof}\ \\
    \underline{$\impliedby$}: Suppose that $\beta=c\alpha^r$, for some $c\in F$ and some integer $r$ which is coprime to $n$. Note that we immediately have the equality and inclusion $F(\beta) = F(c\alpha^r) = F(\alpha^r)\sub F(\alpha)$, thus it suffices to show that $[F(\alpha): F(\alpha^r)] = |\Gal(F(\alpha)/F(\alpha^r))| = 1$. We will show that the Galois group of this extension is trivial.\\

    Suppose that $\sigma\in \Gal(F(\alpha)/F(\alpha^r))$. Let $\zeta$ be a primitive $n^{th}$ root of unity. Noting that $\sigma(\alpha)$ is also a root of $f(x)$ (since $f(x)$ is irreducible), it follows that $\sigma(\alpha) = \zeta^k \alpha$ for some $0\leq k\leq n-1$. Observe that
    \[
      \alpha^r = \sigma(\alpha^r) = (\zeta^k \alpha)^r \implies \alpha^r(\zeta^{kr}-1) = 0 \implies \zeta^{kr}-1 = 0,
    \]
    so $(\zeta^r)^k = 1$. As $r$ is coprime to $n$, we have that $\zeta^r$ is also a primitive $n^{th}$ root of unity, whence $n\divides k$. As $0\leq k \leq n-1$, it must hold that $k = 0$, so $\sigma(\alpha) = \alpha$ whence $\sigma = id$ as desired. \\


    \underline{$\implies$}: Suppose that $F(\alpha) = F(\beta)$. Then $G = \Gal(F(\alpha)/F) = \Gal(F(\beta)/F)$. By Kummer's theorem and irreducibility of $f$ and $g$, $G$ is cyclic of order $n$. Let $G = \cyclic{\sigma}$ and $\zeta$ is a primitive $n^{th}$ root of unity. Then there exist $r,s\in\N$ both coprime to $n$ such that $\sigma(\alpha) = \zeta^r \alpha$ and $\sigma(\beta) = \zeta^s\beta$. As $\zeta^r$ and $\zeta^s$ are both also primitive $n^{th}$ roots of unity, there is some $k\in\N$ coprime to $n$ such that $\zeta^s = (\zeta^r)^k$. Then, we observe that
    \[
      \frac{\sigma(\beta)}{\beta} = \zeta^s = (\zeta^r)^k = \lr{\frac{\sigma(\alpha)}{\alpha}}^k \implies \sigma\lr{\frac{\beta}{\alpha^k}} = \frac{\beta}{\alpha^k}
    \]
    thus $c = \frac{\beta}{\alpha^k}$ is fixed by $G$ so $c\in F$. Thus $\beta = c\alpha^k$ for some $c\in F$ and $k$ coprime to $n$.


  \end{proof}
\end{homeworkProblem}
